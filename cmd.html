const { cmd, commands } = require('../command')

cmd({
    pattern: "menu",
    desc: "Show the interactive command menu.",
    category: "main",
    react: "рҹ“‘",
    filename: __filename
},
async (conn, mek, m, { from, pushname, reply }) => {
    try {
        let menuImg = "https://i.ibb.co/wht9HsYF/20260301-084015-2.jpg"

        let menuText = `*рҹ‘Ӣ HбҙҮКҹКҹбҙҸ ${pushname}*

*QбҙңбҙҮбҙҮЙҙ-RбҙҖкңұКңбҙң-Mбҙ… MбҙҮЙҙбҙң LЙӘкңұбҙӣ*

1пёҸвғЈ бҙҸбҙЎЙҙбҙҮКҖ бҙ„бҙҸбҙҚбҙҚбҙҖЙҙбҙ…кңұ
2пёҸвғЈ ЙўКҖбҙҸбҙңбҙҳ бҙ„бҙҸбҙҚбҙҚбҙҖЙҙбҙ…кңұ
3пёҸвғЈ бҙ…бҙҸбҙЎЙҙКҹбҙҸбҙҖбҙ… бҙ„бҙҸбҙҚбҙҚбҙҖЙҙбҙ…кңұ
4пёҸвғЈ бҙҚбҙҖЙӘЙҙ бҙ„бҙҸбҙҚбҙҚбҙҖЙҙбҙ…кңұ
5пёҸвғЈ бҙ„бҙҸЙҙбҙ бҙҮКҖбҙӣ бҙ„бҙҸбҙҚбҙҚбҙҖЙҙбҙ…кңұ
6пёҸвғЈ кңұбҙҮбҙҖКҖбҙ„Кң бҙ„бҙҸбҙҚбҙҚбҙҖЙҙбҙ…кңұ

*Reply with number Select Other Menu*

> *бҙҳбҙҸбҙЎбҙҮКҖбҙ… КҷКҸ QбҙңбҙҮбҙҮЙҙ КҖбҙҖкңұКңбҙң бҙҸкң°бҙ„*`;

        const sentMsg = await conn.sendMessage(from, {
            image: { url: menuImg },
            caption: menuText
        }, { quoted: mek })

        const msgId = sentMsg.key.id

        const handler = async (update) => {
            try {
                const msg = update.messages[0]
                if (!msg.message?.extendedTextMessage) return

                const userReply = msg.message.extendedTextMessage.text.toLowerCase()
                const contextId = msg.message.extendedTextMessage.contextInfo?.stanzaId

                if (contextId !== msgId) return

                // рҹӣ‘ STOP
                if (userReply === "stop") {
                    conn.ev.off('messages.upsert', handler)
                    return await conn.sendMessage(from, { text: "рҹӣ‘ Menu Closed" })
                }

                let selectedCategory = ""

                if (userReply === '1') selectedCategory = "owner"
                else if (userReply === '2') selectedCategory = "group"
                else if (userReply === '3') selectedCategory = "download"
                else if (userReply === '4') selectedCategory = "main"
                else if (userReply === '5') selectedCategory = "convert"
                else if (userReply === '6') selectedCategory = "search"

                if (!selectedCategory) return

                let cmdList = `рҹ“Ӯ *RASHU ${selectedCategory.toUpperCase()} COMMANDS*\n\n`

                let found = false

                for (let i = 0; i < commands.length; i++) {
                    if (commands[i].category === selectedCategory) {
                        cmdList += `рҹӘ„ рқҗ”рқҗ’рқҗ„ : ${commands[i].pattern}\n`
                        cmdList += `рҹ–ҮпёҸ рқҗғрқҗ„рқҗ’рқҗӮрқҗ‘рқҗҲрқҗҸрқҗ“рқҗҲрқҗҺрқҗҚ : ${commands[i].desc || "No description"}\n\n`
                        found = true
                    }
                }

                if (found) {
                    await conn.sendMessage(from, {
                        image: { url: menuImg },
                        caption: cmdList
                    }, { quoted: msg })
                }

                // вқҢ DO NOT OFF HERE (allow multi use)

            } catch (err) {
                console.log("Menu Handler Error:", err)
            }
        }

        conn.ev.on('messages.upsert', handler)

    } catch (e) {
        console.log(e)
        reply(`Error: ${e}`)
    }
})
